def select(str_or_regex)
  item = items.detect { |i| str_or_regex.is_a?(Regexp) ? str_or_regex.match(i.text) : i.text == str_or_regex }
  raise "Could not locate a radio button matching #{str_or_regex}." unless item

  item.set
end
