def locate_label
  return label unless label.text.empty?

  prev_row = tr(xpath: '(../../preceding-sibling::tr)[last()]')
  lbl = prev_row.labels.first
  while lbl.nil? || lbl.text.empty?
    prev_row = prev_row.tr(xpath: '(../../preceding-sibling::tr)[last()]')
    lbl = prev_row.labels.first
  end
  lbl
end
