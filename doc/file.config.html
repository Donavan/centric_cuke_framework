require 'yaml'

module Helpers
  class Config
    include Singleton
    extend Forwardable
    def_delegator :@data, :[]

    def initialize
      @data = {}
      load
    end

    def load
      Dir.glob(File.join(Nenv.config_path, '*.yml')) do |yml_file|
        @data[File.basename(yml_file, '.yml').to_sym] = YAML.load(File.read(yml_file))
      end
    end
  end
end
